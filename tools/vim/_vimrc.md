Set of custom rules that I use:

```sh
if has("gui_running")
	set guifont=Consolas:h11
	hi normal guifg=white guibg=black
	colorscheme retrobox
endif

set encoding=utf-8

syntax on
set number
set linebreak
set ruler

set background=dark

set expandtab
set tabstop=4
set shiftwidth=4
set autoindent
set cindent
set smartindent
set softtabstop=4

set backspace=indent,eol,start

set noundofile
set nobackup

```
